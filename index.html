📚 1. KELAS BUKU - CONSTRUCTOR DASAR

🔍 KONSEP: Constructor adalah method spesial yang dipanggil saat objek dibuat. Fungsinya untuk menginisialisasi nilai awal atribut.

📌 PENJELASAN: Dalam contoh ini, kita membuat kelas Buku dengan atribut judul, pengarang, dan tahun terbit. Constructor memastikan ketiga data ini selalu diisi saat objek dibuat.

📁 buku.dart DART 
class Buku {
  String judul;
  String pengarang;
  int tahunTerbit;

  // Constructor dengan 3 parameter
  Buku(this.judul, this.pengarang, this.tahunTerbit);

  void info() {
    print('📖 "$judul" oleh $pengarang ($tahunTerbit)');
  }
}

void main() {
  Buku buku1 = Buku('Dart untuk Pemula', 'John Doe', 2024);
  Buku buku2 = Buku('OOP Lanjutan', 'Jane Smith', 2023);
  
  buku1.info();
  buku2.info();
}

🚗 2. KELAS MOBIL - NAMED CONSTRUCTOR

🔍 KONSEP: Named constructor memungkinkan kita memiliki beberapa cara membuat objek. Ini berguna ketika ada beberapa skenario inisialisasi yang berbeda.

📌 PENJELASAN: Kita buat named constructor untuk mobil listrik dan mobil bekas yang punya perilaku berbeda.

📁 mobil.dart DART 
class Mobil {
  String merk;
  int tahun;
  String jenis;
  bool isBekas;

  // Constructor utama
  Mobil(this.merk, this.tahun, this.jenis, this.isBekas);

  // Named constructor untuk mobil listrik baru
  Mobil.listrik(String merk, int tahun)
      : this(merk, tahun, 'Listrik', false);

  // Named constructor untuk mobil bekas (bensin)
  Mobil.bekas(String merk, int tahun)
      : this(merk, tahun, 'Bensin', true);

  void info() {
    String status = isBekas ? 'Bekas' : 'Baru';
    print('🚗 $merk $tahun - $jenis ($status)');
  }
}

void main() {
  Mobil mobil1 = Mobil.listrik('Tesla', 2024);
  Mobil mobil2 = Mobil.bekas('Toyota', 2018);
  
  mobil1.info();
  mobil2.info();
}

🎨 3. KELAS WARNA - CONSTANT CONSTRUCTOR

🔍 KONSEP: Constant constructor membuat objek yang immutable (tidak bisa diubah). Objek dengan nilai yang sama akan menggunakan memory yang sama (canonicalization).

📌 PENJELASAN: Kita buat warna dengan nama dan kode hex, serta warna-warna standar sebagai konstanta.

📁 warna.dart DART 
class Warna {
  final String nama;
  final String kodeHex;

  const Warna(this.nama, this.kodeHex);

  // Warna-warna standar sebagai konstanta
  static const Warna merah = Warna('Merah', '#FF0000');
  static const Warna hijau = Warna('Hijau', '#00FF00');
  static const Warna biru = Warna('Biru', '#0000FF');
}

void main() {
  const warna1 = Warna.merah;
  const warna2 = Warna('Merah', '#FF0000');
  
  print('Warna 1: ${warna1.nama} (${warna1.kodeHex})');
  print('Warna 2: ${warna2.nama} (${warna2.kodeHex})');
  print('Objek sama? ${identical(warna1, warna2)}'); // true
}

📐 4. KELAS TITIK - INITIALIZER LIST

🔍 KONSEP: Initializer list dieksekusi sebelum constructor body. Cocok untuk validasi atau manipulasi nilai awal.

📌 PENJELASAN: Kita validasi bahwa titik tidak boleh di luar batas (0-100) dan otomatis menyesuaikan.

📁 titik.dart DART 
class Titik {
  final int x;
  final int y;

  // Initializer list dengan validasi batas 0-100
  Titik(int x, int y)
      : x = x.clamp(0, 100),
        y = y.clamp(0, 100) {
    print('Titik dibuat: ($x, $y) - (setelah validasi: (${this.x}, ${this.y}))');
  }

  double jarakKe(Titik lain) {
    var dx = x - lain.x;
    var dy = y - lain.y;
    return (dx*dx + dy*dy).toDouble().sqrt();
  }
}

void main() {
  Titik t1 = Titik(25, 30);
  Titik t2 = Titik(150, 200); // Akan dibatasi ke 100,100
  
  print('Jarak: ${t1.jarakKe(t2).toStringAsFixed(2)}');
}

👨‍🎓 5. KELAS MAHASISWA - NAMED CONSTRUCTOR

🔍 KONSEP: Named constructor dengan berbagai sumber data: dari input langsung, dari Map, dan dengan nilai default.

📌 PENJELASAN: Kita tambahkan IPK dan validasi sederhana.

📁 mahasiswa.dart DART 
class Mahasiswa {
  String nama;
  String nim;
  String jurusan;
  double ipk;

  // Constructor utama
  Mahasiswa(this.nama, this.nim, this.jurusan, this.ipk);

  // Dari Map (misal dari API)
  Mahasiswa.dariApi(Map<String, dynamic> data)
      : nama = data['name'],
        nim = data['student_id'],
        jurusan = data['major'] ?? 'Belum ditentukan',
        ipk = (data['gpa'] ?? 0.0).toDouble();

  // Mahasiswa baru dengan IPK default 0.0
  Mahasiswa.baru(String nama, String nim, String jurusan)
      : this(nama, nim, jurusan, 0.0);

  String get predikat {
    if (ipk >= 3.5) return 'Cumlaude';
    if (ipk >= 3.0) return 'Sangat Memuaskan';
    if (ipk >= 2.5) return 'Memuaskan';
    return 'Cukup';
  }

  void info() {
    print('👨‍🎓 $nama ($nim) - $jurusan');
    print('   IPK: $ipk ($predikat)');
  }
}

void main() {
  var data = {'name': 'Budi', 'student_id': '12345', 'gpa': 3.8};
  Mahasiswa m1 = Mahasiswa.dariApi(data);
  Mahasiswa m2 = Mahasiswa.baru('Ani', '67890', 'Informatika');
  
  m1.info();
  m2.info();
}

👤 6. KELAS PERSON - ENKAPSULASI

🔍 KONSEP: Enkapsulasi menyembunyikan data internal dan menyediakan akses melalui method. Field private menggunakan underscore _.

📌 PENJELASAN: Kita tambahkan validasi untuk email dan nomor telepon.

📁 person.dart DART 
class Person {
  String _nama;
  int _umur;
  String _email;
  String _noHP;

  Person(this._nama, this._umur, this._email, this._noHP);

  // Getter
  String get nama => _nama;
  int get umur => _umur;
  String get email => _email;
  String get noHP => _noHP;

  // Setter dengan validasi
  set umur(int nilai) {
    if (nilai > 0 && nilai < 150) {
      _umur = nilai;
    } else {
      print('❌ Umur tidak valid');
    }
  }

  set email(String nilai) {
    if (nilai.contains('@') && nilai.contains('.')) {
      _email = nilai;
    } else {
      print('❌ Email tidak valid');
    }
  }

  set noHP(String nilai) {
    if (nilai.length >= 10 && nilai.length <= 13) {
      _noHP = nilai;
    } else {
      print('❌ Nomor HP tidak valid');
    }
  }

  void info() {
    print('👤 $_nama, $_umur tahun');
    print('   Email: $_email');
    print('   No HP: $_noHP');
  }
}

void main() {
  Person p = Person('Budi', 25, 'budi@email.com', '081234567890');
  p.info();
  
  p.email = 'email-salah'; // Akan gagal validasi
  p.noHP = '123'; // Akan gagal
  p.umur = -5; // Akan gagal
  
  p.email = 'budi.baru@email.com'; // Berhasil
  p.info();
}

📦 7. KELAS PRODUK - VALIDASI SETTER

🔍 KONSEP: Validasi pada setter memastikan data yang masuk sesuai aturan bisnis.

📌 PENJELASAN: Kita tambahkan stok dan validasi diskon.

📁 produk.dart DART 
class Produk {
  String _nama;
  double _harga;
  int _stok;
  double _diskon = 0;

  Produk(this._nama, this._harga, this._stok);

  // Getter
  String get nama => _nama;
  double get harga => _harga;
  int get stok => _stok;
  double get diskon => _diskon;

  // Setter dengan validasi
  set nama(String value) {
    if (value.length >= 3) {
      _nama = value;
    } else {
      print('❌ Nama minimal 3 karakter');
    }
  }

  set harga(double value) {
    if (value > 0) {
      _harga = value;
    } else {
      print('❌ Harga harus positif');
    }
  }

  set stok(int value) {
    if (value >= 0) {
      _stok = value;
    } else {
      print('❌ Stok tidak boleh negatif');
    }
  }

  set diskon(double value) {
    if (value >= 0 && value <= 100) {
      _diskon = value;
    } else {
      print('❌ Diskon harus 0-100%');
    }
  }

  double get hargaSetelahDiskon {
    return _harga * (1 - _diskon / 100);
  }

  void info() {
    print('📦 $_nama');
    print('   Harga: Rp${_harga.toStringAsFixed(0)}');
    if (_diskon > 0) {
      print('   Diskon: $_diskon%');
      print('   Harga setelah diskon: Rp${hargaSetelahDiskon.toStringAsFixed(0)}');
    }
    print('   Stok: $_stok');
  }
}

void main() {
  Produk p = Produk('Laptop', 10000000, 5);
  p.info();
  
  p.diskon = 20; // Berhasil
  p.stok = -1; // Gagal
  p.nama = 'PC'; // Gagal (terlalu pendek)
  p.nama = 'Laptop Gaming'; // Berhasil
  
  p.info();
}

🔐 8. KELAS USER - SISTEM LOGIN

🔍 KONSEP: Sistem login dengan enkapsulasi password dan validasi keamanan.

📌 PENJELASAN: Kita tambahkan role (admin/user) dan login attempt.

📁 user.dart DART 
class User {
  String _username;
  String _password;
  String _email;
  String _role;
  int _loginAttempts = 0;
  bool _isLocked = false;

  User(this._username, this._password, this._email, this._role);

  // Getter
  String get username => _username;
  String get email => _email;
  String get role => _role;
  bool get isLocked => _isLocked;
  int get loginAttempts => _loginAttempts;

  bool login(String username, String password) {
    if (_isLocked) {
      print('🔒 Akun terkunci! Hubungi admin.');
      return false;
    }

    if (username == _username && password == _password) {
      _loginAttempts = 0;
      print('✅ Login berhasil sebagai $_role');
      return true;
    } else {
      _loginAttempts++;
      print('❌ Login gagal. Percobaan ke-$_loginAttempts');
      
      if (_loginAttempts >= 3) {
        _isLocked = true;
        print('🔒 Akun terkunci karena 3x gagal login!');
      }
      return false;
    }
  }

  bool gantiPassword(String lama, String baru) {
    if (lama != _password) {
      print('❌ Password lama salah');
      return false;
    }

    if (baru.length < 8) {
      print('❌ Password baru minimal 8 karakter');
      return false;
    }

    if (baru == lama) {
      print('❌ Password baru harus berbeda');
      return false;
    }

    _password = baru;
    print('✅ Password berhasil diubah');
    return true;
  }

  void info() {
    print('👤 @$_username ($_role)');
    print('   Email: $_email');
    print('   Status: ${_isLocked ? "Terkunci" : "Aktif"}');
  }
}

void main() {
  User admin = User('admin', 'rahasia123', 'admin@system.com', 'ADMIN');
  
  admin.info();
  admin.login('admin', 'salah'); // Gagal 1
  admin.login('admin', 'salah2'); // Gagal 2
  admin.login('admin', 'salah3'); // Gagal 3 -> terkunci
  admin.login('admin', 'rahasia123'); // Sudah terkunci
}

🏦 9. KELAS REKENING BANK - TRANSAKSI

🔍 KONSEP: Simulasi rekening bank dengan fitur deposit, tarik, transfer, dan history transaksi.

📌 PENJELASAN: Kita tambahkan history transaksi dan biaya admin.

📁 rekening.dart DART 
class RekeningBank {
  String _nomorRekening;
  String _namaPemilik;
  double _saldo;
  List<Map<String, dynamic>> _history = [];

  RekeningBank(this._nomorRekening, this._namaPemilik, this._saldo);

  // Getter
  String get nomorRekening => _nomorRekening;
  String get namaPemilik => _namaPemilik;
  double get saldo => _saldo;
  List<Map<String, dynamic>> get history => 
      List.unmodifiable(_history);

  void _tambahHistory(String jenis, double jumlah, {String? keterangan}) {
    _history.add({
      'waktu': DateTime.now().toString(),
      'jenis': jenis,
      'jumlah': jumlah,
      'saldo_setelah': _saldo,
      'keterangan': keterangan ?? '-',
    });
  }

  bool deposit(double jumlah) {
    if (jumlah <= 0) {
      print('❌ Jumlah deposit harus positif');
      return false;
    }

    _saldo += jumlah;
    _tambahHistory('DEPOSIT', jumlah);
    print('✅ Deposit Rp${jumlah.toStringAsFixed(0)} berhasil');
    return true;
  }

  bool tarik(double jumlah) {
    if (jumlah <= 0) {
      print('❌ Jumlah tarik harus positif');
      return false;
    }

    if (_saldo < jumlah) {
      print('❌ Saldo tidak mencukupi (Saldo: Rp${_saldo.toStringAsFixed(0)})');
      return false;
    }

    _saldo -= jumlah;
    _tambahHistory('TARIK', jumlah);
    print('✅ Tarik Rp${jumlah.toStringAsFixed(0)} berhasil');
    return true;
  }

  bool transfer(RekeningBank tujuan, double jumlah) {
    if (jumlah <= 0) {
      print('❌ Jumlah transfer harus positif');
      return false;
    }

    const double biayaAdmin = 2500;
    double total = jumlah + biayaAdmin;

    if (_saldo < total) {
      print('❌ Saldo tidak cukup untuk transfer + biaya admin');
      return false;
    }

    _saldo -= total;
    tujuan._saldo += jumlah;
    
    _tambahHistory('TRANSFER KELUAR', jumlah, 
        keterangan: 'Ke: ${tujuan._namaPemilik}');
    tujuan._tambahHistory('TRANSFER MASUK', jumlah,
        keterangan: 'Dari: $_namaPemilik');

    print('✅ Transfer Rp${jumlah.toStringAsFixed(0)} ke ${tujuan._namaPemilik} berhasil');
    print('   Biaya admin: Rp$biayaAdmin');
    return true;
  }

  void info() {
    print('\n🏦 REKENING: $_nomorRekening');
    print('   Nama: $_namaPemilik');
    print('   Saldo: Rp${_saldo.toStringAsFixed(0)}');
  }

  void tampilkanHistory() {
    print('\n📋 HISTORY TRANSAKSI $_namaPemilik:');
    for (var t in _history.reversed) {
      print('   ${t['waktu'].toString().substring(0,19)}');
      print('   ${t['jenis']}: Rp${t['jumlah']}');
      if (t['keterangan'] != '-') {
        print('   ${t['keterangan']}');
      }
      print('   ---');
    }
  }
}

void main() {
  RekeningBank rekening1 = RekeningBank('001-01', 'Budi', 1000000);
  RekeningBank rekening2 = RekeningBank('001-02', 'Ani', 500000);

  rekening1.info();
  rekening2.info();

  rekening1.deposit(500000);
  rekening1.tarik(200000);
  rekening1.transfer(rekening2, 300000);

  rekening1.info();
  rekening2.info();

  rekening1.tampilkanHistory();
  rekening2.tampilkanHistory();
}
